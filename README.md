```
  const findLst = func(s, a, b) => {

    const aIndex = s.lastIndexOf(a);
    const bIndex = s.lastIndexOf(b);

    if (aIndex !== -1) {
        if (bIndex === -1) {
            return aIndex;
        } else {
            return Math.max(aIndex, bIndex);
        }
    } else if (bIndex !== -1) {
        return bIndex;
    } else {
        return -1;
    }
}
```
